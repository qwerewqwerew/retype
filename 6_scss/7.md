# 1. if

```scss
body {
	margin: 50px;
}
@mixin textAndBgColor($textColor, $bgColor) {
	color: $textColor;
	background-color: $bgColor;
}
@mixin theme($mood) {
	@if $mood == 'light' {
		@include textAndBgColor(#333, #ff0);
	} @else if $mood == 'dark' {
		@include textAndBgColor(#fff, #000);
	} @else {
		@include textAndBgColor(#f00, #aaa);
	}
}
body {
	@include theme('light');
}
body {
	@include theme('dark');
}
body {
	@include theme('mid');
}
```
